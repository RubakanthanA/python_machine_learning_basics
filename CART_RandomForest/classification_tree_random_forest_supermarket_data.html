Fit the classification tree model and make predictions

In[7]:
# Initialize the classification tree model 
# mach ein Vorhersagemodell um Testdatenset vorherzusagen
# DecisionTreeClassifier: erstellt einen Entscheidugnsbaum Klassifikationsmodell
# random_state: legt fest, dass Entscheidungsbaum jedes Mal auf die gleiche Weise erstellt wird
# max_depth=5: legt fest, dass der Baum maximal 5 Ebenen haben kann

clf = DecisionTreeClassifier(random_state=42, 
                             max_depth=5)


# Train the classification tree model - unverndert
# Das Modell trainiert (fit) das Entscheidungsbaum-Modell clf, sodass es die Beziehung zwischen
# den Eingabedaten (X_train) und den Zielwerten (y_train) versteht.

clf = clf.fit(X_train, y_train)

# Make model predictions - unverndert
# verwendet das trainierte Modell, um Vorhersagen fr die Testdaten zu machen
# y_pred: speichert die Vorhersagen

y_pred = clf.predict(X_test)

Fit the Random Forest Classifier

In[]:
# Initialize the random forest classifier
# RandomForestClassifier: erstellt ein Random Forest Klassifikationsmodell
# dieser kombiniert mehrere Entscheidungsbume, um eine genauere Vorhersage zu treffen
# n_estimators: hier werden 500 Bume verwendet, um die Vorhersageleistung zu verbessern
# random_state: legt fest, dass der Zufallsfaktor reproduzierbar ist, immer der gleiche Wert
# max_depth=5: legt fest, dass der Baum maximal 5 Ebenen haben kann, dies hilft Overfitting zu vermeiden
rfc = RandomForestClassifier(n_estimators=500, random_state=42, max_depth=5)

# Train the classification tree model
# fit: trainiert den random forest classifier, um Muster zu erkennen
rfc = rfc.fit(X2_train, y2_train)

# Predict the target variable
# Modell verwendet die gelernte Daten oben die Testdaten, um Vorhersagen zu machen
y_pred_rf = rfc.predict(X2_test)

# Erstellt die ROC-Kurve und berechnet den AUC-Wert
# Die ROC-Kurve (Receiver Operating Characteristic) zeigt die Leistung eines Klassifikationsmodells, indem sie die True-Positive-Rate gegen die False-Positive-Rate fr verschiedene Schwellenwerte auftrgt.
# Der AUC-Wert (Area Under the Curve) misst die Flche unter der ROC-Kurve. Ein AUC-Wert von 1,0 bedeutet eine perfekte Klassifikation, whrend 0,5 auf zuflliges Raten hindeutet.

# plt.figure: erstellt eine neue Abbildung (6 Breite, 6 Hhe)
