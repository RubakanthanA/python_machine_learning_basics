Fit the classification tree model and make predictions¶

In [7]:
# Initialize the classification tree model 
# mach ein Vorhersagemodell um Testdatenset vorherzusagen
# DecisionTreeClassifier: erstellt einen Entscheidugnsbaum Klassifikationsmodell
# random_state: legt fest, dass Entscheidungsbaum jedes Mal auf die gleiche Weise erstellt wird
# max_depth=5: legt fest, dass der Baum maximal 5 Ebenen haben kann

clf = DecisionTreeClassifier(random_state=42, 
                             max_depth=5)


# Train the classification tree model - unverändert
# Das Modell trainiert (fit) das Entscheidungsbaum-Modell clf, sodass es die Beziehung zwischen
# den Eingabedaten (X_train) und den Zielwerten (y_train) versteht.

clf = clf.fit(X_train, y_train)

# Make model predictions - unverändert
# verwendet das trainierte Modell, um Vorhersagen für die Testdaten zu machen
# y_pred: speichert die Vorhersagen

y_pred = clf.predict(X_test)

# Confusion matrix - Überblick wie oft das Modell eine korrekte Vorhersage gemacht hat.

Fit the Random Forest Classifier¶

In [ ]:
# Initialize the random forest classifier
# RandomForestClassifier: erstellt ein Random Forest Klassifikationsmodell
# dieser kombiniert mehrere Entscheidungsbäume, um eine genauere Vorhersage zu treffen
# n_estimators: hier werden 500 Bäume verwendet, um die Vorhersageleistung zu verbessern
# random_state: legt fest, dass der Zufallsfaktor reproduzierbar ist, immer der gleiche Wert
# max_depth=5: legt fest, dass der Baum maximal 5 Ebenen haben kann, dies hilft Overfitting zu vermeiden
rfc = RandomForestClassifier(n_estimators=500, random_state=42, max_depth=5)

# Train the classification tree model
# fit: trainiert den random forest classifier, um Muster zu erkennen
rfc = rfc.fit(X2_train, y2_train)

# Predict the target variable
# Modell verwendet die gelernte Daten oben die Testdaten, um Vorhersagen zu machen
y_pred_rf = rfc.predict(X2_test)

# Crosstab
# brand: tatsächliche Werte der Zielvariablen
# df_pred: es wird ein DataFrame erstellt, das die tatsächlichen Werte und die vorhergesagten Werte enthält
# Marken in zwei Spalten (brand und brand_predicted) nebeneinander anzeigt.
# reset_index(drop=True): setzt den Index zurück, so dass die Zeilen von 0 bis n nummeriert sind
# df_pred: zeigt die tatsächlichen und vorhergesagten Marken an
print('Brand versus predicted brand:', '\n')
df_pred = pd.DataFrame(data={'brand': y2_test, 
                             'brand_predicted': y_pred_rf}).reset_index(drop=True)

# Plot ROC curve and calculate AUC
# Erstellt die ROC-Kurve und berechnet den AUC-Wert
# Die ROC-Kurve (Receiver Operating Characteristic) zeigt die Leistung eines Klassifikationsmodells, indem sie die True-Positive-Rate gegen die False-Positive-Rate für verschiedene Schwellenwerte aufträgt.
# Der AUC-Wert (Area Under the Curve) misst die Fläche unter der ROC-Kurve. Ein AUC-Wert von 1,0 bedeutet eine perfekte Klassifikation, während 0,5 auf zufälliges Raten hindeutet.

# plt.figure: erstellt eine neue Abbildung (6 Breite, 6 Höhe)
# ax = plt.gca(): Speichert das aktuelle Achsenobjekt (ax), sodass später an die RocCurveDisplay-Funktion übergeben kann.

# RocCurveDisplay.from_estimator: Diese Funktion erzeugt die ROC-Kurve basierend auf einem Modell.
# rf_disp: erstellt eine ROC-Kurve basierend auf einem Modell
# X2_test und y2_test: Die Testdaten und die tatsächlichen Labels für die Zielvariable, die für die Berechnung der ROC-Kurve verwendet werden.
# ax=ax: Legt fest, dass die Kurve in das zuvor erstellte Achsenobjekt ax gezeichnet wird.
# alpha=0.8: Bestimmt die Transparenz der Kurve, liegt üblicherweise zwischen 0 und 1.
#   (alpha=0 bedeutet vollständig transparent (unsichtbar) und lpha=1 bedeutet vollständig deckend (keine Transparenz))
# Farbe der Kurve: darkred
# plt.grid: fügt ein Gitter hinzu
# plt.show: zeigt die fertige Grafik an

plt.figure(figsize=(6,6))
ax = plt.gca()
rfc_disp = RocCurveDisplay.from_estimator(rfc, 
                                          X2_test, 
                                          y2_test, 
                                          ax=ax,
                                          alpha=0.8,
                                          c="darkred")
plt.grid()
plt.show()
